# Move model to the selected device
model = model.to(device)
print(f'The device in use is {device}.\n')

criterion = nn.NLLLoss()
optimizer = optim.Adam(model.classifier.parameters(), lr=0.0001)


epochs = 15
steps = 0
batches = 0
print_every = 1
patience = 10  # Number of epochs to wait for improvement
min_val_loss = float("inf")
wait = 0
training_losses = []

running_loss = running_accuracy = 0
validation_losses, training_losses = [],[]

for e in range(epochs):
    model.train()  # Set the model to training mode
    running_loss = 0

    for inputs, labels in trainloader:  # Assuming you have a trainloader defined
        steps += 1
        # Move input and label tensors to the default device
        inputs, labels = inputs.to(device), labels.to(device)

        optimizer.zero_grad()

        logps = model.forward(inputs)
        loss = criterion(logps, labels)

        # Apply L2 regularization
        l2_reg = torch.tensor(0.0).to(device)
        for param in model.parameters():
            l2_reg += torch.norm(param, p=2)
        loss += 0.001 * l2_reg
        loss.backward()
        optimizer.step()

        running_loss += loss.item()

        if steps % print_every == 0:
            test_loss = 0
            accuracy = 0
            model.eval()
            with torch.no_grad():
                for inputs, labels in testloader:
                    inputs, labels = inputs.to(device), labels.to(device)
                    logps = model.forward(inputs)
                    batch_loss = criterion(logps, labels)

                    test_loss += batch_loss.item()

                    train_loss = running_loss / len(trainloader)
                    training_losses.append(train_loss)

                    # Calculate accuracy
                    ps = torch.exp(logps)
                    top_p, top_class = ps.topk(1, dim=1)
                    equals = top_class == labels.view(*top_class.shape)
                    accuracy += torch.mean(equals.type(torch.FloatTensor)).item()

            print(f"Epoch {e+1}/{epochs}.. "
                  f"Train loss: {running_loss/print_every:.3f}.. "
                  f"Test loss: {test_loss/len(testloader):.3f}.. "
                  f"Test accuracy: {accuracy/len(testloader)*100:.3f}%")
            running_loss = 0
            model.train()  # Set the model back to training mode

            # Early stopping check
            if train_loss < min_val_loss:
               min_val_loss = train_loss
               wait = 0
            else:
               wait += 1

            if wait >= patience:
               print("Early stopping triggered. Training stopped.")
               break

def load_checkpoint(filepath):
    checkpoint = torch.load(filepath)

    if checkpoint['structure'] == 'vgg16':
        model = models.vgg16(pretrained = True)

        for param in model.parameters():
            param.requires_grad = False

        model.classifier = checkpoint['classifier']

        model.load_state_dict(checkpoint['state_dict'])
    return model

from PIL import Image
import torchvision.transforms as transforms
import torch

# Define class labels based on your model's classes
class_labels = ["NORMAL", "PNEUMONIA"]

def process_image(image_path):
    # Load the image
    img = Image.open(image_path)

    # Check the number of channels in the image
    if img.mode != 'RGB':
        # Convert grayscale to RGB
        img = img.convert('RGB')

    # Define image transformations (resize, center-crop, convert to tensor, normalize)
    preprocess = transforms.Compose([
        transforms.Resize(256),
        transforms.CenterCrop(224),
        transforms.ToTensor(),
        transforms.Normalize(mean=[0.485, 0.456, 0.406], std=[0.229, 0.224, 0.225]),
    ])

    # Apply transformations to the image
    img = preprocess(img).unsqueeze(0)  # Add a batch dimension

    return img

def predict(image_path, model, topk=1):
    # Process the image
    img = process_image(image_path)

    # Move the image tensor to the device
    img = img.to(device)

    # Set the model to evaluation mode
    model.eval()

    # Perform the forward pass to get probabilities
    with torch.no_grad():
        output = model(img)

    # Calculate probabilities and class indices
    probabilities = torch.exp(output)
    top_probs, top_indices = probabilities.topk(topk)

    # Map indices to class labels using class_labels list
    top_classes = [class_labels[idx] for idx in top_indices.cpu().numpy()[0]]

    return top_probs.cpu().numpy()[0], top_classes

from PIL import Image
import matplotlib.pyplot as plt
import torchvision.transforms as transforms
import torch

# Define class names
class_names = ['NORMAL', 'PNEUMONIA']

# Create a mapping of class names to indices
class_to_idx = {class_name: idx for idx, class_name in enumerate(class_names)}

def predict(image_path, model):
    # Load the image
    img = Image.open(image_path)

    # Ensure that the image has 3 channels (RGB)
    img = img.convert('RGB')

    # Define image transformations (resize, center-crop, convert to tensor, normalize)
    preprocess = transforms.Compose([
        transforms.Resize(256),
        transforms.CenterCrop(224),
        transforms.ToTensor(),
        transforms.Normalize(mean=[0.485, 0.456, 0.406], std=[0.229, 0.224, 0.225]),
    ])

    # Apply transformations to the image
    img = preprocess(img).unsqueeze(0)  # Add a batch dimension

    # Move the image tensor to the device
    img = img.to(device)

    # Set the model to evaluation mode
    model.eval()

    # Perform the forward pass to get probabilities
    with torch.no_grad():
        output = model(img)

    # Calculate probabilities and class indices
    probabilities = torch.exp(output)
    top_probs, top_indices = probabilities.topk(1)

    # Convert indices to class labels
    top_class_idx = top_indices.cpu().numpy()[0][0]
    top_class_name = class_names[top_class_idx]

    return top_probs.item(), top_class_name

def plot_image_with_prediction(image_path, model, class_names):
    # Get the probability and predicted class
    prob, predicted_class = predict(image_path, model)

    # Load the image
    img = Image.open(image_path)

    # Create a figure with subplots
    fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(10, 4))

    # Plot the image
    ax1.imshow(img)
    ax1.axis('off')
    ax1.set_title('Image')

    # Plot the probability bar chart
    ax2.barh(['NORMAL', predicted_class], [1 - prob, prob], color=['green', 'red'])
    ax2.set_xlim(0, 1)
    ax2.set_xlabel('Probability')
    ax2.set_title('Predicted Class')

    plt.tight_layout()
    plt.show()

# Example usage:
image_path = "/content/drive/MyDrive/chest_xray/test/PNEUMONIA/person100_bacteria_477.jpeg"
plot_image_with_prediction(image_path, model, class_names)
